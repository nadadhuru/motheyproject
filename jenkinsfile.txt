#!/bin/
def path = "motheyproject"

pipeline
{
agent any

stages
    {
	stage($buildchoice)
	{
		steps
			{
			dir(path)
				{
						if($$buildchoice == "Compile")
						{
							bat 'mvn compile'
						}
						else if($$buildchoice == "Run Test")
						{
							bat 'mvn test'
						}
						else
						{
						bat 'mvn deploy'
						}
				
				}
			}
	}
     }
   }
