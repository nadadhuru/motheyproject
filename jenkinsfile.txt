def path = "motheyproject"

pipeline
{
agent any

stages
    {
	stage("Maven Build")
	{
		steps
			{
			script{
			
			dir(path)
				{
						if($buildchoice == "Compile")
						{
							bat 'mvn compile'
						}
						else if($buildchoice == "Run Test")
						{
							bat 'mvn test'
						}
						else
						{
						bat 'mvn deploy'
						}
				
				}
			}
		}
	}
     }
   }
