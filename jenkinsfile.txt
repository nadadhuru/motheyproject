#!/bin/
def path = "motheyproject"

pipeline
{
agent any

if($$buildchoice == "Compile")
{
  stages
    {
	stage("Compile")
		{
		steps
			{
				step
					{
						dir(path)
						bat 'mvn compile'
					}
			}
		}
	}
}
else if($$buildchoice == "Run Test")
{
  stages
    {
	stage("Test")
		{
			steps
				{
					step
						{
							dir(path)
							bat 'mvn test'
						}
				}
		}
	}
}
else
{
	stage("Deploy")
	{
		steps
			{
				step
					{
						dir(path)
						bat 'mvn deploy'
					}
			}
	}
}
    
}
